组建初始化 

await IAuthApi.init(
    loginFutureBuilder: (LoginType e, final Map<String, dynamic>? request) async {
  switch (e) {
    case LoginType.pwd:
      // case LoginType.email:
      return (UserApi.login(LoginRequestBean.fromJson(request ?? {})).then((value) {
        UserApi.userObs.value = UserInfoBean.fromJson(value.toJson());
        return value.toJson();
      }));
    default:
      return Future.error(const CheckFailedResponse<TokenBean>(
        statusMessage: '未知登录类型',
      ));
  }
});
